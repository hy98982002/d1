```ts
// src/stores/courseStore.ts
import { defineStore } from 'pinia'

export type CourseLevel = 'beginner' | 'intermediate' | 'advanced' | 'deep-intermediate'

export interface Course {
  id: string
  title: string
  cover: string
  level: CourseLevel
  price: number | null
  students: number
  isVipOnly?: boolean        // ⭐ 会员专属课程（深度进阶课）
  comingSoon?: boolean       // ⭐ 即将上线占位卡
  tags?: string[]
}

export const useCourseStore = defineStore('courseStore', {
  state: () => ({
    courses: [
      // ============================
      // 🔵 Beginner（免费）
      // ============================
      {
        id: 'ps-basic',
        title: 'Photoshop 入门基础',
        cover: '/images/photoshop-basic-cover-480.webp',
        level: 'beginner',
        price: 0,
        students: 1200,
        tags: ['Photoshop', 'AIGC']
      },

      // ============================
      // 🟢 Intermediate（普通进阶课）
      // ============================
      {
        id: 'logo-analysis',
        title: 'logo 分析',
        cover: '/images/logo-analysis-cover-480.webp',
        level: 'intermediate',
        price: 699,
        students: 600
      },
      {
        id: 'ps-environment',
        title: 'Photoshop 场景环境课程',
        cover: '/images/ps-env-cover-480.webp',
        level: 'intermediate',
        price: 499,
        students: 700
      },

      // ============================
      // ⭐ 深度进阶 Intermediate+（会员10门）
      // ⭐ 可单买但价格更高；课程卡显示「深度进阶」，难度仍保持中级体系
      // ============================
      {
        id: 'ai-brand-system',
        title: 'AI 品牌视觉设计系统',
        cover: '/images/logo-design-membership-intermediate-cover-480.webp',
        level: 'deep-intermediate',  // ← 新难度等级（属于中级，但更深）
        price: 899,
        students: 0,
        isVipOnly: true
      },
      {
        id: 'ai-game-scene',
        title: 'AI 游戏场景设计',
        cover: '/images/ai-game-membership-cover-480.webp',
        level: 'deep-intermediate',
        price: 999,
        students: 0,
        isVipOnly: true
      },

      // ============================
      // ⏳ 即将上线（广告占位卡）
      // 自动显示在会员专区底部
      // ============================
      {
        id: 'coming-1',
        title: 'AI 动态海报创作实战（即将上线）',
        cover: '',
        level: 'deep-intermediate',
        price: null,
        students: 0,
        isVipOnly: true,
        comingSoon: true
      }
    ] as Course[]
  }),

  getters: {
    beginnerCourses: state => state.courses.filter(c => c.level === 'beginner'),

    intermediateCourses: state =>
      state.courses.filter(c => c.level === 'intermediate'),

    deepIntermediateCourses: state =>
      state.courses.filter(c => c.level === 'deep-intermediate'),

    vipCourses: state =>
      state.courses.filter(c => c.isVipOnly === true),

    comingSoonCourses: state =>
      state.courses.filter(c => c.comingSoon === true)
  }
})
```
